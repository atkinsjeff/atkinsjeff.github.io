Line Graphs

With data of this sort, you may want to present it in the form of a line graph. We can do this with geom_line().

# Just the lines
ggplot(df.tidy, aes(x = year, y = dbh, color = species))+
  geom_line(size = 2)+
  ylab("DBH (cm)")+
  xlab("Year")+
  theme_bw()+
  scale_color_manual(values=c("#999999", "#E69F00", "#56B4E9", "#009E73"), 
                       name="Species",
                       breaks=c("oak", "pine", "maple", "birch"),
                       labels=c("Oak", "Pine", "Maple", "Birch"))+
  scale_x_continuous(breaks=seq(2005, 2017, 4))+
  ylim(0, 35)+
 theme(legend.justification=c(1,0), 
       legend.position=c(1,0))

ggtitle("DBH by Species from 2005-2017")

But is this how you would want to see this data?

require(plyr)
## Loading required package: plyr
require(dplyr)
## Loading required package: dplyr
##

##

## The following objects are masked from 'package:base':
## 
##     intersect, setdiff, setequal, union
df2 %>% 
  group_by(year, species, landform) %>%
  summarise(Mean = mean(growth),
            SD = sd(growth)) -> m.growth
  

# Now we can plot something more meaningful
ggplot(m.growth, aes(x = year, y = Mean, group = species, color = species))+
  geom_line(size = 1)+
  geom_point(size = 2)+
  geom_errorbar(aes(ymin = Mean - SD, ymax = Mean + SD), width = 0.1)+
  ylab("Diameter Growth (cm)")+
  xlab("")+
  theme(legend.position = "bottom")+
ggtitle("DBH by Species from 2015-2017")+
  facet_wrap(~ landform, ncol = 2)

Bar Charts

A bar chart may be a straghtforward way to do this as well

ggplot(m.growth, aes(x = year, y = Mean, fill = species))+
  geom_bar(stat = "identity")+
  geom_errorbar(aes(ymin = Mean - SD, ymax = Mean + SD), width = 0.1)+
  facet_grid(species ~ landform)+  
  ylab("Diameter Growth (cm)")+
  xlab("")+
  theme(legend.position = "none")+
ggtitle("DBH by Species from 2015-2017")
